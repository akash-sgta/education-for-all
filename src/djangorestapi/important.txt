------------------- server setup --------------------------

Test gateway:
>>>
uwsgi --http :8000 --wsgi-file test.py
<<<

Test wsgi file:
>>>
uwsgi --http :8000 --module djangorestapi.wsgi
<<<

Nginx Configs:
/etc/nginx/sites-available/djangorestapi.conf
>>>
upstream django {
    server  unix:///home/akashsgta/repositories/education-for-all/src/djangorestapi/djangorestapi.sock;
}
server {
    listen  80;
    charset utf-8;

    client_max_body_size    75M;

    location    /media {
        alias   /home/akashsgta/repositories/education-for-all/src/djangorestapi/media;
    }
    location    /static {
        alias   /home/akashsgta/repositories/education-for-all/src/djangorestapi/static;
    }

    location    / {
        uwsgi_pass  django;
        include /home/akashsgta/repositories/education-for-all/src/djangorestapi/uwsgi_params;
    }

}
<<<
sudo /etc/init.d/nginx restart
